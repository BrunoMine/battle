# Sistema de Loot

# Sistema de loots aleatorios de itens (requer mod treasurer)
# System for random items loots (require treasurer mod)
# Nodes:
# default:chest
battle_enable_loot (Sistema de loot de itens) bool false

# Isso registra os itens comuns do proprio Minetest para aparecerem nos loots
# This registers the common items from Minetest itself to appear in the loots
battle_load_default_loot_itens (Itens comuns para loot) bool true

# Isso registra itens do mod 3d_armor para aparecerem nos loots
# This registers the armors from 3d_armor mod itself to appear in the loots
battle_load_armor_loot_itens (Loot para 3d_armor) bool true

# Isso registra itens do mod armatesla para aparecerem nos loots
# This registers the armors from armatesla mod itself to appear in the loots
battle_load_armatesla_loot_itens (Loot para Armatesla) bool true

# Isso registra itens do mod cannons para aparecerem nos loots
# This registers the armors from cannons mod itself to appear in the loots
battle_load_cannons_loot_itens (Loot para Cannons) bool true

# Esse Numero define o numero do loot que é incrementado 
# a cada nova partida para sortear novos loots
# This number defines the number of loot 
# that is incremented with each new match to make new loots
battle_loot_number_control (Numero de Controle do Loot) int 1
